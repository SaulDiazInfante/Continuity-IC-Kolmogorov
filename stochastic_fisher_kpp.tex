Let $\mathcal{H} = L^2 (0,1)$. We consider the stochastic Fisher-KPP 
    equation in the interval $[0, 1]$
    \begin{equation}
        \label{eqn:fisher-kpp}
        \begin{aligned}
            d X(t, \xi) &= 
                \left[
                    \nu 
                    \partial_{\xi} ^ 2 X(t, \xi)
                    +
                    X(t, \xi) (1 -X(t, \xi) )
                \right]
                dt
                +
                dW(t, \xi),
            \\
            X(t, 0) &= X(t, 1) =0, \quad t>0, 
            \\
            X(0, \xi) &= x(\xi), \quad x\in \mathcal{H}
        \end{aligned}
    \end{equation}
